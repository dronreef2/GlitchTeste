## üîÑ Pipeline CI/CD

### GitHub Actions Workflow

```yaml
# .github/workflows/deploy.yml
name: DevOps Platform Deploy

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
          cache: 'npm'
      
      - name: Install dependencies
        run: npm ci
      
      - name: Run tests
        run: npm test
      
      - name: Run linting
        run: npm run lint
      
      - name: Security audit
        run: npm audit

  build:
    needs: test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Build Docker image
        run: docker build -t devops-platform .
      
      - name: Run container tests
        run: docker run --rm devops-platform npm test

  deploy-staging:
    needs: [test, build]
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/develop'
    steps:
      - name: Deploy to Staging
        run: |
          curl -X POST ${{ secrets.STAGING_WEBHOOK_URL }} \
            -H "Authorization: Bearer ${{ secrets.DEPLOY_TOKEN }}" \
            -d '{"environment": "staging", "branch": "develop"}'

  deploy-production:
    needs: [test, build]
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
      - name: Deploy to Production
        run: |
          curl -X POST ${{ secrets.PRODUCTION_WEBHOOK_URL }} \
            -H "Authorization: Bearer ${{ secrets.DEPLOY_TOKEN }}" \
            -d '{"environment": "production", "branch": "main"}'
```

### Script de Deploy Automatizado

```bash
#!/bin/bash
# scripts/deploy.sh

set -e

ENVIRONMENT=$1
VERSION=$2
BACKUP_DIR="/opt/backups"
APP_DIR="/opt/devops-platform"

echo "üöÄ Iniciando deploy para $ENVIRONMENT..."

# Fun√ß√£o de logging
log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1"
}

# Fun√ß√£o de backup
create_backup() {
    log "üì¶ Criando backup..."
    timestamp=$(date +%Y%m%d_%H%M%S)
    backup_file="$BACKUP_DIR/backup_${ENVIRONMENT}_${timestamp}.tar.gz"
    
    tar -czf "$backup_file" -C "$APP_DIR" . 2>/dev/null || true
    log "‚úÖ Backup criado: $backup_file"
}

# Fun√ß√£o de health check
health_check() {
    local url=$1
    local max_attempts=30
    local attempt=1
    
    log "üîç Executando health check..."
    
    while [ $attempt -le $max_attempts ]; do
        if curl -sf "$url/api/health" > /dev/null; then
            log "‚úÖ Health check passou!"
            return 0
        fi
        
        log "‚è≥ Tentativa $attempt/$max_attempts falhou, aguardando..."
        sleep 10
        ((attempt++))
    done
    
    log "‚ùå Health check falhou ap√≥s $max_attempts tentativas"
    return 1
}

# Fun√ß√£o de rollback
rollback() {
    log "‚è™ Executando rollback..."
    
    # Encontrar √∫ltimo backup
    latest_backup=$(ls -t $BACKUP_DIR/backup_${ENVIRONMENT}_*.tar.gz | head -n1)
    
    if [ -n "$latest_backup" ]; then
        log "üì¶ Restaurando de: $latest_backup"
        tar -xzf "$latest_backup" -C "$APP_DIR"
        systemctl restart devops-platform
        log "‚úÖ Rollback conclu√≠do"
    else
        log "‚ùå Nenhum backup encontrado para rollback"
        return 1
    fi
}

# Fun√ß√£o principal de deploy
deploy() {
    log "üîÑ Parando servi√ßos..."
    systemctl stop devops-platform || true
    
    create_backup
    
    log "üì• Baixando nova vers√£o..."
    docker pull "registry.company.com/devops-platform:$VERSION"
    
    log "üîÑ Atualizando configura√ß√£o..."
    docker-compose -f "docker-compose.$ENVIRONMENT.yml" up -d
    
    log "‚è≥ Aguardando inicializa√ß√£o..."
    sleep 30
    
    if [ "$ENVIRONMENT" = "production" ]; then
        health_url="https://api.company.com"
    else
        health_url="https://staging-api.company.com"
    fi
    
    if health_check "$health_url"; then
        log "üéâ Deploy conclu√≠do com sucesso!"
        
        # Notificar Slack
        curl -X POST -H 'Content-type: application/json' \
            --data "{\"text\":\"‚úÖ Deploy de $ENVIRONMENT conclu√≠do com sucesso!\"}" \
            "$SLACK_WEBHOOK_URL"
    else
        log "‚ùå Deploy falhou, executando rollback..."
        rollback
        
        # Notificar falha
        curl -X POST -H 'Content-type: application/json' \
            --data "{\"text\":\"‚ùå Deploy de $ENVIRONMENT falhou! Rollback executado.\"}" \
            "$SLACK_WEBHOOK_URL"
        
        exit 1
    fi
}

# Validar par√¢metros
if [ -z "$ENVIRONMENT" ] || [ -z "$VERSION" ]; then
    echo "Uso: $0 <environment> <version>"
    echo "Exemplo: $0 production v1.2.3"
    exit 1
fi

# Executar deploy
deploy
```

### Script de Monitoramento Cont√≠nuo

```bash
#!/bin/bash
# scripts/monitor.sh

SERVICES=("api" "worker" "monitor" "webssh" "webftp")
ALERT_WEBHOOK="https://hooks.slack.com/services/YOUR/SLACK/WEBHOOK"
LOG_FILE="/var/log/devops-platform/monitor.log"

log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"
}

check_service() {
    local service=$1
    local port=$2
    
    if curl -sf "http://localhost:$port/health" > /dev/null 2>&1; then
        log "‚úÖ $service est√° funcionando"
        return 0
    else
        log "‚ùå $service n√£o est√° respondendo"
        return 1
    fi
}

check_resources() {
    # CPU
    cpu_usage=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | awk -F'%' '{print $1}')
    if (( $(echo "$cpu_usage > 80" | bc -l) )); then
        log "‚ö†Ô∏è Alto uso de CPU: ${cpu_usage}%"
        alert "CPU usage high: ${cpu_usage}%"
    fi
    
    # Mem√≥ria
    mem_usage=$(free | grep Mem | awk '{printf("%.2f", $3/$2 * 100.0)}')
    if (( $(echo "$mem_usage > 85" | bc -l) )); then
        log "‚ö†Ô∏è Alto uso de mem√≥ria: ${mem_usage}%"
        alert "Memory usage high: ${mem_usage}%"
    fi
    
    # Disco
    disk_usage=$(df -h / | awk 'NR==2 {print $5}' | sed 's/%//')
    if [ "$disk_usage" -gt 90 ]; then
        log "‚ö†Ô∏è Alto uso de disco: ${disk_usage}%"
        alert "Disk usage high: ${disk_usage}%"
    fi
}

alert() {
    local message=$1
    curl -X POST -H 'Content-type: application/json' \
        --data "{\"text\":\"üö® $message\"}" \
        "$ALERT_WEBHOOK"
}

restart_service() {
    local service=$1
    log "üîÑ Reiniciando $service..."
    
    systemctl restart "devops-platform-$service"
    sleep 10
    
    if check_service "$service" "${service_ports[$service]}"; then
        log "‚úÖ $service reiniciado com sucesso"
        alert "$service was restarted successfully"
    else
        log "‚ùå Falha ao reiniciar $service"
        alert "Failed to restart $service - manual intervention required"
    fi
}

# Portas dos servi√ßos
declare -A service_ports=(
    ["api"]="3000"
    ["worker"]="3001"
    ["monitor"]="9090"
    ["webssh"]="2222"
    ["webftp"]="3333"
)

# Loop principal de monitoramento
while true; do
    log "üîç Executando verifica√ß√µes de sa√∫de..."
    
    for service in "${SERVICES[@]}"; do
        if ! check_service "$service" "${service_ports[$service]}"; then
            restart_service "$service"
        fi
    done
    
    check_resources
    
    log "‚è≥ Aguardando pr√≥xima verifica√ß√£o..."
    sleep 300  # 5 minutos
done
```

### Script de Backup Automatizado

```bash
#!/bin/bash
# scripts/backup.sh

BACKUP_DIR="/opt/backups"
RETENTION_DAYS=30
S3_BUCKET="company-devops-backups"
ENVIRONMENT=${NODE_ENV:-development}

create_backup() {
    local timestamp=$(date +%Y%m%d_%H%M%S)
    local backup_name="devops-platform_${ENVIRONMENT}_${timestamp}"
    local backup_file="$BACKUP_DIR/${backup_name}.tar.gz"
    
    echo "üì¶ Criando backup: $backup_name"
    
    # Backup de arquivos
    tar -czf "$backup_file" \
        --exclude='node_modules' \
        --exclude='*.log' \
        --exclude='.git' \
        /opt/devops-platform
    
    # Backup de banco de dados (se existir)
    if [ ! -z "$DATABASE_URL" ]; then
        echo "üóÑÔ∏è Fazendo backup do banco de dados..."
        pg_dump "$DATABASE_URL" > "$BACKUP_DIR/${backup_name}_db.sql"
        gzip "$BACKUP_DIR/${backup_name}_db.sql"
    fi
    
    # Upload para S3
    if [ ! -z "$AWS_ACCESS_KEY_ID" ]; then
        echo "‚òÅÔ∏è Enviando backup para S3..."
        aws s3 cp "$backup_file" "s3://$S3_BUCKET/$(basename $backup_file)"
        
        if [ -f "$BACKUP_DIR/${backup_name}_db.sql.gz" ]; then
            aws s3 cp "$BACKUP_DIR/${backup_name}_db.sql.gz" "s3://$S3_BUCKET/$(basename ${backup_name}_db.sql.gz)"
        fi
    fi
    
    echo "‚úÖ Backup conclu√≠do: $backup_file"
}

cleanup_old_backups() {
    echo "üßπ Limpando backups antigos (>$RETENTION_DAYS dias)..."
    
    find "$BACKUP_DIR" -name "devops-platform_*.tar.gz" -mtime +$RETENTION_DAYS -delete
    find "$BACKUP_DIR" -name "*_db.sql.gz" -mtime +$RETENTION_DAYS -delete
    
    # Limpeza no S3
    if [ ! -z "$AWS_ACCESS_KEY_ID" ]; then
        aws s3 ls "s3://$S3_BUCKET/" | \
        awk '$1 < "'$(date -d "$RETENTION_DAYS days ago" +%Y-%m-%d)'" {print $4}' | \
        xargs -I {} aws s3 rm "s3://$S3_BUCKET/{}"
    fi
    
    echo "‚úÖ Limpeza conclu√≠da"
}

# Criar diret√≥rio de backup se n√£o existir
mkdir -p "$BACKUP_DIR"

# Executar backup e limpeza
create_backup
cleanup_old_backups

echo "üéâ Processo de backup finalizado!"
```

# alert_rules.yml
groups:
  - name: devops-platform-alerts
    rules:
      - alert: HighCPUUsage
        expr: (100 - (avg by (instance) (rate(node_cpu_seconds_total{mode="idle"}[5m])) * 100)) > 80
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "High CPU usage on {{ $labels.instance }}"
          description: "CPU usage is above 80% for more than 5 minutes"

      - alert: HighMemoryUsage
        expr: (1 - (node_memory_MemAvailable_bytes / node_memory_MemTotal_bytes)) * 100 > 85
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "High memory usage on {{ $labels.instance }}"
          description: "Memory usage is above 85% for more than 5 minutes"

      - alert: ServiceDown
        expr: up == 0
        for: 1m
        labels:
          severity: critical
        annotations:
          summary: "Service {{ $labels.job }} is down"
          description: "Service has been down for more than 1 minute"

      - alert: HighErrorRate
        expr: rate(http_requests_total{status=~"5.."}[5m]) > 0.1
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "High error rate in {{ $labels.job }}"
